## 1. 概述

NavLink 主站是一个现代化的导航页面应用，提供以下核心功能：

- 🔍 全局搜索（支持多搜索引擎）
- 🎨 主题定制系统
- 📊 链接分类管理
- 🔥 热门推荐区域
- 💬 AI 聊天助手
- 📱 响应式设计
- 🔐 可选的用户认证

**核心功能**：

**关键代码**：

```typescript
import React, { useState, useEffect } from 'react';
import { ConfigProvider, useConfig } from '@/shared/context/ConfigContext';
import TopNavbar from '@/shared/components/layout/TopNavbar';
import SearchHero from './components/home/SearchHero';
import PromoArea from './components/home/PromoArea';
import CategorySection from './components/home/CategorySection';
import RightWidgets from './components/home/RightWidgets';
import Footer from './components/home/Footer';

function AppContent() {
  const { config, isLoaded, isAuthenticated, login, logout } = useConfig();
  const [activeCategory, setActiveCategory] = useState<string>('');
  const [showSearchModal, setShowSearchModal] = useState(false);
  const [collapsed, setCollapsed] = useState(false);

  // 动态更新 Favicon
  useEffect(() => {
    const link = document.querySelector("link[rel~='icon']") || document.createElement('link');
    link.href = config.logoUrl || 'default-icon-data-url';
    document.head.appendChild(link);
  }, [config.logoUrl]);

  // 监听键盘快捷键
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      const shortcut = config.searchShortcut || 'Cmd+K';
      // 解析快捷键并触发搜索
      if (matchesShortcut(e, shortcut)) {
        e.preventDefault();
        setShowSearchModal(true);
      }
    };
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [config.searchShortcut]);

  if (!isLoaded) {
    return <LoadingScreen />;
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-50 to-gray-100">
      <TopNavbar 
        config={config}
        isAuthenticated={isAuthenticated}
        onUserClick={handleUserIconClick}
        onLogout={logout}
        onSearchClick={() => setShowSearchModal(true)}
      />
      
      <SearchHero config={config} />
      <PromoArea config={config} />
      
      <div className="max-w-screen-2xl mx-auto px-4 py-8 flex gap-8">
        <Sidebar 
          config={config}
          activeCategory={activeCategory}
          onScrollTo={scrollTo}
          collapsed={collapsed}
          toggleCollapsed={() => setCollapsed(!collapsed)}
        />
        
        <main className="flex-1">
          <CategorySection config={config} />
        </main>
        
        <RightWidgets config={config} />
      </div>
      
      <Footer config={config} />
      
      {/* 模态框 */}
      {showSearchModal && (
        <SearchModal

          onClose={() => setShowSearchModal(false)}
        />
      )}
    </div>
  );
}

export default function App() {
  return (
    <ConfigProvider>
      <AppContent />
    </ConfigProvider>
  );
}
```

```typescript
function SearchHero({ config }: SearchHeroProps) {
  const [searchEngine, setSearchEngine] = useState(config.searchEngines[0]);
  const [searchQuery, setSearchQuery] = useState('');

  const handleSearch = (e: React.FormEvent) => {
    e.preventDefault();
    if (!searchQuery.trim()) return;
    
    const searchUrl = searchEngine.urlPattern + encodeURIComponent(searchQuery);
    window.open(searchUrl, '_blank');
  };

  return (
    <div 
      className="relative h-[400px] bg-cover bg-center"
      style={{
        backgroundImage: config.backgroundImage 
          ? `url(${config.backgroundImage})` 
          : `linear-gradient(135deg, ${config.hero.backgroundColor || '#667eea'} 0%, #764ba2 100%)`
      }}
    >
      <div className="absolute inset-0 bg-black/30" />
      
      <div className="relative z-10 container mx-auto px-4 h-full flex flex-col justify-center items-center text-white">
        <h1 className="text-5xl font-bold mb-4">{config.hero.title}</h1>
        <p className="text-xl mb-8 opacity-90">{config.hero.subtitle}</p>
        
        <form onSubmit={handleSearch} className="w-full max-w-2xl">
          <div className="flex gap-2">
            <select 
              value={searchEngine.id}
              onChange={(e) => setSearchEngine(config.searchEngines.find(se => se.id === e.target.value)!)}
              className="px-4 py-3 rounded-lg bg-white/90 text-gray-800"
            >
              {config.searchEngines.map(se => (
                <option key={se.id} value={se.id}>{se.name}</option>
              ))}
            </select>
            
            <input
              type="text"
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
              placeholder={searchEngine.placeholder}
              className="flex-1 px-6 py-3 rounded-lg bg-white/90 text-gray-800 focus:outline-none focus:ring-2 focus:ring-white"
            />
            
            <button 
              type="submit"
              className="px-8 py-3 bg-blue-600 hover:bg-blue-700 rounded-lg font-medium transition"
            >
              搜索
            </button>
          </div>
        </form>
        
        {/* 热门搜索 */}
        <div className="mt-6 flex gap-3 flex-wrap justify-center">
          {config.hero.hotSearchLinks.map(link => (
            <a
              key={link.title}
              href={link.url}
              target="_blank"
              rel="noopener noreferrer"
              className="px-4 py-1 bg-white/20 hover:bg-white/30 rounded-full text-sm transition"
            >
              {link.title}
            </a>
          ))}
        </div>
      </div>
    </div>
  );
}
```

**使用方式**：

```typescript

```

**实现示例**：

```typescript
function CategorySection({ config }: CategorySectionProps) {
  const { isAuthenticated } = useConfig();
  const [searchQuery, setSearchQuery] = useState('');
  
  // 过滤隐藏分类
  const visibleCategories = config.categories.filter(cat => 
    isAuthenticated || !cat.hidden
  );
  
  // 搜索过滤
  const filteredCategories = useMemo(() => {
    if (!searchQuery) return visibleCategories;
    
    return visibleCategories.map(category => {
      const filteredItems = category.items?.filter(item =>
        item.title.toLowerCase().includes(searchQuery.toLowerCase()) ||
        item.description.toLowerCase().includes(searchQuery.toLowerCase())
      );
      
      const filteredSubCategories = category.subCategories?.map(subCat => ({
        ...subCat,
        items: subCat.items.filter(item =>
          item.title.toLowerCase().includes(searchQuery.toLowerCase())
        )
      })).filter(subCat => subCat.items.length > 0);
      
      return {
        ...category,
        items: filteredItems,
        subCategories: filteredSubCategories
      };
    }).filter(category => 
      category.items?.length || category.subCategories?.length
    );
  }, [visibleCategories, searchQuery]);

  return (
    <div className="space-y-8">
      {/* 搜索栏 */}
      <div className="sticky top-20 z-10 bg-white/80 backdrop-blur-sm p-4 rounded-lg">
        <input
          type="text"
          value={searchQuery}
          onChange={(e) => setSearchQuery(e.target.value)}
          placeholder="搜索链接..."
          className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
        />
      </div>
      
      {/* 分类列表 */}
      {filteredCategories.map(category => (
        <CategoryCard key={category.id} category={category} />
      ))}
    </div>
  );
}

function CategoryCard({ category }: { category: Category }) {
  const [activeSubCategory, setActiveSubCategory] = useState(
    category.subCategories?.[0]?.name
  );

  return (
    <div id={category.id} className="bg-white rounded-xl shadow-sm p-6">
      {/* 分类标题 */}
      <h2 className="text-2xl font-bold text-gray-800 mb-4 flex items-center gap-3">
        <i className={`${category.icon} text-blue-600`}></i>
        {category.name}
      </h2>
      
      {/* 子分类标签 */}
      {category.subCategories && (
        <div className="flex gap-2 mb-4 overflow-x-auto">
          {category.subCategories.map(subCat => (
            <button
              key={subCat.name}
              onClick={() => setActiveSubCategory(subCat.name)}
              style={{
                backgroundColor: activeSubCategory === subCat.name 
                  ? subCat.color || '#3b82f6' 
                  : undefined
              }}
              className={`
                px-4 py-2 rounded-lg transition whitespace-nowrap
                ${activeSubCategory === subCat.name 
                  ? 'text-white' 
                  : 'bg-gray-100 text-gray-600 hover:bg-gray-200'}
              `}
            >
              {subCat.name}
            </button>
          ))}
        </div>
      )}
      
      {/* 链接网格 */}
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {(category.subCategories 
          ? category.subCategories.find(sc => sc.name === activeSubCategory)?.items 
          : category.items
        )?.map(item => (
          <LinkCard key={item.id} item={item} />
        ))}
      </div>
    </div>
  );
}

function LinkCard({ item }: { item: LinkItem }) {
  return (
    <a
      href={item.url}
      target="_blank"
      rel="noopener noreferrer"
      className="group p-4 bg-gray-50 rounded-lg hover:bg-blue-50 hover:shadow-md transition"
    >
      <div className="flex items-start gap-3">
        {/* 图标 */}
        <div 
          className="w-10 h-10 rounded flex items-center justify-center flex-shrink-0"
          style={{ backgroundColor: item.color || '#3b82f6' }}
        >
          {item.icon ? (
            <i className={`${item.icon} text-white`}></i>
          ) : (
            <span className="text-white font-bold">{item.title[0]}</span>
          )}
        </div>
        
        {/* 内容 */}
        <div className="flex-1 min-w-0">
          <h3 className="font-medium text-gray-800 group-hover:text-blue-600 transition truncate">
            {item.title}
          </h3>
          <p className="text-sm text-gray-500 line-clamp-2 mt-1">
            {item.description}
          </p>
          
          {/* 健康状态 */}
          {item.health && (
            <div className="flex items-center gap-2 mt-2">
              <div className={`
                w-2 h-2 rounded-full
                ${item.health.isHealthy ? 'bg-green-500' : 'bg-red-500'}
              `} />
              <span className="text-xs text-gray-400">
                {item.health.isHealthy 
                  ? `${item.health.responseTime}ms` 
                  : '不可用'}
              </span>
            </div>
          )}
        </div>
      </div>
    </a>
  );
}
```

```typescript
function RightWidgets({ config }: RightWidgetsProps) {
  const [hotTopics, setHotTopics] = useState<HotTopic[]>([]);
  const [githubTrending, setGithubTrending] = useState<GithubRepo[]>([]);

  // 加载热门话题
  useEffect(() => {
    const sources = config.rightSidebar.hotTopics;
    sources.forEach(async (source) => {
      const response = await fetch(source.apiUrl);
      const data = await response.json();
      setHotTopics(prev => [...prev, ...data.slice(0, source.limit || 10)]);
    });
  }, [config.rightSidebar.hotTopics]);

  return (
    <aside className="hidden xl:block w-80 space-y-6">
      {/* 个人资料 */}
      <ProfileCard profile={config.rightSidebar.profile} />
      
      {/* 热门话题 */}
      <HotTopicsCard topics={hotTopics} />
      
      {/* GitHub 趋势 */}
      <GithubTrendingCard 
        repos={githubTrending}
        config={config.rightSidebar.githubTrending}
      />
    </aside>
  );
}
```

```typescript
function scrollTo(id: string) {
  const element = document.getElementById(id);
  if (element) {
    const offset = 100; // 导航栏高度
    const elementPosition = element.getBoundingClientRect().top;
    const offsetPosition = elementPosition + window.pageYOffset - offset;

    window.scrollTo({

      behavior: 'smooth'
    });
  }
}
```

### A. 常用工具函数

**颜色处理**：

```typescript
// 判断颜色是否为浅色
function isLightColor(color: string): boolean {
  const hex = color.replace('#', '');
  const r = parseInt(hex.substring(0, 2), 16);
  const g = parseInt(hex.substring(2, 4), 16);
  const b = parseInt(hex.substring(4, 6), 16);
  return (r * 299 + g * 587 + b * 114) / 1000 >= 128;
}
```

**防抖函数**：

```typescript
function debounce<T extends (...args: any[]) => any>(
  func: T,
  wait: number
): (...args: Parameters<T>) => void {
  let timeout: ReturnType<typeof setTimeout>;
  
  return function executedFunction(...args: Parameters<T>) {
    clearTimeout(timeout);
    timeout = setTimeout(() => func(...args), wait);
  };
}
```
